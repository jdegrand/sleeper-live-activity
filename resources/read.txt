Inside your ActivityView or ContentState views:

struct SleeperLiveActivityView: View {
    let context: ActivityViewContext<SleeperLiveActivityAttributes>

    var body: some View {
        HStack {
            // Local asset
            Image("team_logo")
                .resizable()
                .scaledToFit()
                .frame(width: 40, height: 40)
                .clipShape(Circle())

            // Remote image
            AsyncImage(url: URL(string: "https://example.com/avatar.png")) { image in
                image.resizable()
            } placeholder: {
                ProgressView()
            }
            .frame(width: 40, height: 40)
            .clipShape(Circle())

            Text(context.state.teamName)
        }
    }
}
